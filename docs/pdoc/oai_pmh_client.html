 66            which matches repositories (like arXiv) that reject second-level timestamps
 67            in from/until parameters.

 74    def _format_datestamp(self, dt: Datestamp) -> str:
 75        """
 76        Formats a datetime object into an OAI-PMH datestamp string.
 77        """
 78        if isinstance(dt, str):
 79            return dt
 80        if dt.tzinfo is None:
 81            # If the datetime object is naive, assume it's in UTC.
 82            dt = dt.replace(tzinfo=timezone.utc)
 83        # If the datetime object is aware, convert it to UTC.
 84        dt = dt.astimezone(timezone.utc)
 85        if self.datestamp_granularity == "YYYY-MM-DD":
 86            return dt.strftime("%Y-%m-%d")
 87        # Default / fallback to second-level granularity
 88        return dt.strftime("%Y-%m-%dT%H:%M:%SZ")

 90    def _request(self, verb: str, **kwargs) -> etree._Element:
 91        """
 92        Makes a request to the OAI-PMH repository and returns the parsed XML.
 93
 94        :param verb: The OAI-PMH verb.
 95        :param kwargs: Additional request parameters.
 96        :return: The parsed XML response.
 97        """
 98        params = {"verb": verb}
 99        # Filter out None values so they aren't included in the query
100        for key, value in kwargs.items():
101            if value is not None:
102                params[key] = value
103
104        if self.use_post:
105            response = self._client.post(self.base_url, data=params)
106        else:
107            response = self._client.get(self.base_url, params=params)
108
109        response.raise_for_status()
110        xml = etree.fromstring(response.content)
111
112        error = xml.find("oai:error", namespaces=NS)
113        if error is not None:
114            code = error.get("code", "")
115            message = error.text or ""
116            exception_class = OAI_ERROR_MAP.get(code, OAIError)
117            raise exception_class(message)
118
119        return xml

147        """
